/@ 
   Test file for Project#3 compiler
   This file demonstrates basic language features 
@/

/^ Include helper functions
include "toinclude";

/^ Type declarations and variable declarations
Imw globalVar;
SIMw negativeNum;
Chj letter;
Series message;
IMwf pi;
SIMwf negativeFloat;

/^ Structure definition
Loli Person {
    Imw age;
    Series name;
    IMwf height;
};

/^ Function declaration with parameters
Imw add(Imw a, Imw b) {
    Turnback a + b;
}

/^ Function without return value
NOReturn printMessage(Series msg) {
    /^ This function doesn't return anything
    /^ Print operation would be handled by the runtime
}

/^ Main function with various statements
Imw main() {
    /^ Variable declarations
    Imw x, y, z;
    IMwf result;
    Series greeting;
    Loli Person person1;
    
    /^ Assignment operations
    x = 10;
    y = 20;
    z = add(x, y);
    
    /^ Conditional statement
    IfTrue (z > 25) {
        result = z * 1.5;
    } Otherwise {
        result = z / 2.0;
    }
    
    /^ Loop statement
    Imw counter;
    counter = 0;
    RepeatWhen (counter < 5) {
        counter = counter + 1;
        
        IfTrue (counter == 3) {
            OutLoop;
        }
    }
    
    /^ Another loop style
    Reiterate (Imw i = 0; i < 10; i = i + 1) {
        IfTrue (i % 2 == 0) {
            /^ Skip even numbers
            RepeatWhen;
        }
    }
    
    /^ Structure access
    person1.age = 25;
    person1->name = "John Doe";
    person1->height = 175.5;
    
    /^ Return statement
    Turnback 0;
}
